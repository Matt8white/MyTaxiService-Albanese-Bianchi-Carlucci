\newcounter{UseCaseCounter}
\newcounter{ScenarioCounter}
\newcounter{MockupCounter}
\newcounter{SequenceCounter}

\newcommand{\usecase}[6]{
	\refstepcounter{UseCaseCounter}
	\begin{tcolorbox}[
		tab2,
		tabularx*={\arrayrulewidth0.5mm}{Y|X},
		title=Use Case \arabic{UseCaseCounter},
		boxrule=2pt
	]
		Name & #1 \\ \hline
		Description & #2 \\ \hline
		Primary Actor & #3 \\ \hline
		Basic Flow & #4 \\ \hline
		Alternate Flows & #5 \\ \hline
		Exception & #6 
	\end{tcolorbox}
}

\newcommand{\scenario}[3]{
	\refstepcounter{ScenarioCounter}
	\label{scenario:#1}
	\begin{tcolorbox}[
		tab2,
		title=Scenario \arabic{ScenarioCounter},
		boxrule=2pt
	]
		\textbf{#2}
		\tcblower
		#3
	\end{tcolorbox}
}

\newcommand{\mockup}[2]{
    \refstepcounter{MockupCounter}
    \begin{figure}[H]
        \centering
        \includegraphics[scale=0.5]{mockup_#1.jpg}
        \caption*{Mockup \arabic{MockupCounter}: #2}
        \label{mockup:#1}
    \end{figure}
}

\newcommand{\sequence}[2]{
    \refstepcounter{SequenceCounter}
    \pagebreak
    \vfill
    \begin{figure}[H]
        \centering
        \includegraphics[scale=0.5]{Sequence#1.png}
        \caption*{Sequence Diagram \arabic{SequenceCounter}: #2}
        \label{sequence:#1}
    \end{figure}
    \pagebreak
}
